email_text = """\ From: %s To: %s Subject: %s %s """ %(sent_from, ", ".join(to), subject, body)
try: conn_ssl = smtplib.SMTP_SSL('smtp.gmail.com', 465) conn_ssl.ehlo() conn_ssl.login(gmail_user, gmail_password) conn_ssl.sendmail(gmail_user, to, email_text) conn_ssl.close()
